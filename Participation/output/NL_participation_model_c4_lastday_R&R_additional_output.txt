        model$BHHHse <- rep(NA, length(model$estimate))

            `p(1-sided)` = 1 * (1 - stats::pnorm(abs(model$estimate/tmp))), 
            `Bootstrap.t.rat.(1)` = (model$estimate - 1)/tmp,

            tmp <- tmp[, -grep("t.rat.(1", colnames(tmp), fixed = TRUE)]
        output <- cbind(output, tmp)
    }

    if (printClassical == FALSE) 
        dropcolumns = c(dropcolumns, 2, 3, 4, 5, 6)
    if (printBHHH == FALSE) 
        dropcolumns = c(dropcolumns, 12, 13, 14, 15, 16)
    if (printT1 == FALSE) 
        dropcolumns = c(dropcolumns, 5, 6, 10, 11, 15, 16, 20, 
            21)
    if (printPVal == FALSE) 
        dropcolumns = c(dropcolumns, 4, 6, 9, 11, 14, 16, 19, 
            21)

        output = output[, -dropcolumns, drop = FALSE]
